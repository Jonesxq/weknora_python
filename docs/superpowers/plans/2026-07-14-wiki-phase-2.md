# Wiki 阶段二实施计划

> **面向执行代理：** 必须使用 `superpowers:executing-plans` 或 `superpowers:subagent-driven-development` 按任务执行；每项均遵守测试先行和提交边界。

**目标：** 在当前 `app/` 结构中实现真实 Celery/Redis/Outbox 基础设施和 fake 上游驱动的 Wiki 基础 Map/Reduce 摄取流水线。

**架构：** PostgreSQL 中的 pending-op 是业务队列真源，Outbox 负责可靠触发 Celery，Redis token 锁保证单 KB 串行。知识/chunk、模型和 finalizing 通过端口隔离；阶段二提供 JSON fixture fake，Worker 在模型调用完成后使用短事务原子提交页面、链接、日志、任务完成和 finalization 释放。

**技术栈：** Python 3.12、FastAPI、Pydantic v2、SQLAlchemy 2 Async、PostgreSQL 16、Alembic、Celery、Redis、Tenacity、pytest、pytest-asyncio。

```python
@dataclass(frozen=True, slots=True)
class CelerySettings:
    broker_url: str
    result_backend: str
    task_always_eager: bool = False

    @classmethod
    def from_env(cls) -> "CelerySettings":
        return cls(
            broker_url=os.getenv("GRAPH_CELERY_BROKER_URL", "redis://127.0.0.1:6379/0"),
            result_backend=os.getenv(
                "GRAPH_CELERY_RESULT_BACKEND", "redis://127.0.0.1:6379/1"
            ),
            task_always_eager=os.getenv(
                "GRAPH_CELERY_TASK_ALWAYS_EAGER", "false"
            ).casefold() == "true",
        )


def create_celery_app(settings: CelerySettings) -> Celery:
    app = Celery(
        "graph",
        broker=settings.broker_url,
        backend=settings.result_backend,
        include=["app.wiki.tasks.wiki_tasks"],
    )
    app.conf.update(
        task_serializer="json",
        result_serializer="json",
        accept_content=["json"],
        enable_utc=True,
        timezone="UTC",
        task_always_eager=settings.task_always_eager,
        task_eager_propagates=True,
    )
    return app


celery_app = create_celery_app(CelerySettings.from_env())
```

```python
class WikiPendingOp(Base):
    __tablename__ = "wiki_pending_ops"
    __table_args__ = (
        UniqueConstraint(
            "tenant_id", "knowledge_base_id", "knowledge_id", "op", "op_version",
            name="uq_wiki_pending_ops_version",
        ),
        Index(
            "ix_wiki_pending_ops_scope_claim",
            "tenant_id", "knowledge_base_id", "claimed_at", "enqueued_at",
        ),
    )
    id: Mapped[UUID] = mapped_column(UUID(as_uuid=True), primary_key=True, default=_uuid)
    tenant_id: Mapped[int] = mapped_column(BigInteger, nullable=False)
    knowledge_base_id: Mapped[UUID] = mapped_column(UUID(as_uuid=True), nullable=False)
    knowledge_id: Mapped[str] = mapped_column(String(255), nullable=False)
    op: Mapped[str] = mapped_column(String(32), nullable=False, default="ingest")
    op_version: Mapped[str] = mapped_column(String(255), nullable=False)
    payload: Mapped[dict[str, Any]] = mapped_column(JSONB, nullable=False, default=dict)
    fail_count: Mapped[int] = mapped_column(Integer, nullable=False, default=0)
    enqueued_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), nullable=False, server_default=func.now()
    )
    claimed_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))
    claim_token: Mapped[UUID | None] = mapped_column(UUID(as_uuid=True))
```

```python
def rebuild_source_content(chunks: Sequence[SourceChunk], *, max_chars: int = 32768) -> str:
    ordered = sorted(chunks, key=lambda item: (item.chunk_index, item.start_at, item.id))
    parts: list[str] = []
    for chunk in ordered:
        parts.extend(value.strip() for value in (chunk.text, chunk.ocr_text, chunk.image_caption) if value.strip())
    return "\n".join(parts)[:max_chars]


def has_meaningful_text(content: str) -> bool:
    without_images = re.sub(r"!\[[^\]]*\]\([^)]*\)", "", content)
    return len("".join(without_images.split())) >= 10
```

```python
def build_claim_pending_ops_statement(
    scope: WikiScope,
    *,
    limit: int,
    stale_before: datetime,
) -> Select[tuple[WikiPendingOp]]:
    return (
        select(WikiPendingOp)
        .where(
            WikiPendingOp.tenant_id == scope.tenant_id,
            WikiPendingOp.knowledge_base_id == scope.knowledge_base_id,
            or_(
                WikiPendingOp.claimed_at.is_(None),
                WikiPendingOp.claimed_at <= stale_before,
            ),
        )
        .order_by(WikiPendingOp.enqueued_at, WikiPendingOp.id)
        .limit(limit)
        .with_for_update(skip_locked=True)
    )
```

领取事务还必须按本批 `claim_token` 写入一条恢复 Outbox，`available_at` 为 `claimed_at + claim_timeout`。正常结果事务或完整失败释放会把该恢复事件标记为已处理；Worker 在提交前退出时，恢复事件会在 claim 过期后重新触发批次。重新领取 stale claim 时同时取消旧 token 的恢复事件。

入队使用 PostgreSQL `insert().on_conflict_do_nothing()`；三个表写入后按唯一键查询完整 `EnqueueRecord`：

```python
import hashlib
import json
from uuid import UUID


def build_outbox_dedup_key(
    tenant_id: int,
    knowledge_base_id: UUID,
    event_type: str,
    knowledge_id: str,
    op_version: str,
) -> str:
    canonical = json.dumps(
        {
            "tenant_id": tenant_id,
            "knowledge_base_id": str(knowledge_base_id),
            "event_type": event_type,
            "knowledge_id": knowledge_id,
            "op_version": op_version,
        },
        sort_keys=True,
        separators=(",", ":"),
        ensure_ascii=True,
    )
    return hashlib.sha256(canonical.encode("ascii")).hexdigest()


async def enqueue(
    self,
    scope: WikiScope,
    knowledge: SourceKnowledge,
    payload: dict[str, object],
    *,
    delay_seconds: int = 30,
) -> EnqueueRecord:
    event_type = "wiki.batch.trigger"
    dedup_key = build_outbox_dedup_key(
        scope.tenant_id,
        scope.knowledge_base_id,
        event_type,
        knowledge.id,
        knowledge.op_version,
    )
    async with self._session_factory() as session, session.begin():
        await session.execute(
            postgresql.insert(WikiPendingOp)
            .values(
                tenant_id=scope.tenant_id,
                knowledge_base_id=scope.knowledge_base_id,
                knowledge_id=knowledge.id,
                op="ingest",
                op_version=knowledge.op_version,
                payload=payload,
            )
            .on_conflict_do_nothing(constraint="uq_wiki_pending_ops_version")
        )
        await self._finalization.register(
            session,
            FinalizationRequest.from_knowledge(scope, knowledge),
        )
        await session.execute(
            postgresql.insert(TaskOutbox)
            .values(
                tenant_id=scope.tenant_id,
                knowledge_base_id=scope.knowledge_base_id,
                event_type=event_type,
                dedup_key=dedup_key,
                payload={
                    "tenant_id": scope.tenant_id,
                    "knowledge_base_id": str(scope.knowledge_base_id),
                },
                available_at=datetime.now(UTC) + timedelta(seconds=delay_seconds),
            )
            .on_conflict_do_nothing(
                constraint="uq_task_outbox_scope_event_dedup"
            )
        )
        await session.flush()
        return await self.get_enqueue_record(session, scope, knowledge)
```

`SqlFinalizationPort` 只操作 marker，并通过 `RETURNING` 给出是否首次登记/释放：

```python
class SqlFinalizationPort:
    async def register(
        self, session: AsyncSession, request: FinalizationRequest
    ) -> bool:
        result = await session.execute(
            postgresql.insert(WikiFinalizationMarker)
            .values(**request.model_dump())
            .on_conflict_do_nothing(
                constraint="uq_wiki_finalization_markers_attempt"
            )
            .returning(WikiFinalizationMarker.id)
        )
        return result.scalar_one_or_none() is not None

    async def release(
        self, session: AsyncSession, request: FinalizationRequest
    ) -> bool:
        result = await session.execute(
            update(WikiFinalizationMarker)
            .where(
                WikiFinalizationMarker.tenant_id == request.tenant_id,
                WikiFinalizationMarker.knowledge_base_id == request.knowledge_base_id,
                WikiFinalizationMarker.knowledge_id == request.knowledge_id,
                WikiFinalizationMarker.attempt == request.attempt,
                WikiFinalizationMarker.subtask_name == request.subtask_name,
                WikiFinalizationMarker.released_at.is_(None),
            )
            .values(released_at=datetime.now(UTC))
            .returning(WikiFinalizationMarker.id)
        )
        return result.scalar_one_or_none() is not None
```

- [ ] **步骤 4：实现原子页面结果写入**

`find_existing_pages()` 返回冻结的 `ExistingPageRecord(page_id, version, page)`；模型调用只使用其中的 `page` 快照。`apply_results()` 同时接收 `completed_op_ids`、`failed_op_ids` 和逐 slug 的 `expected_pages`，并在一个事务中完成以下操作：

1. 锁定活跃页面，按 `page_id + version` 校验已有页面快照；期望不存在的 slug 在提交时仍必须不存在，否则抛出 `PageConflict`。
2. 新页面创建 `WikiPage(status="draft", version=1)`；已有页面只在可见字段变化时 `version += 1`。
3. `source_refs` 和 aliases 使用稳定去重，不接受客户端字段。
4. 使用 `SqlAlchemyPageStore.replace_page_links()` 重建链接投影。
5. 全部页面写完后统一设置 `status="published"`。
6. 添加一条 `WikiLogEntry(operation_id=operation_id, action="wiki_ingest_batch")`。
7. 释放完成操作的 marker 并删除完成 op；失败 op 在同一事务中执行 `fail_count += 1` 并清除 claim。任意 ID 不属于同一非空 claim token 时抛出 `ClaimLost` 并整体回滚。
8. 仍有 pending 时写 5 秒后的 follow-up Outbox。completed 与 failed 均为空时只安全返回，不凭空调度。

保留的独立 `release_failed()` 也必须把失败释放和 follow-up Outbox 放在同一事务中，避免进程在两步之间退出造成队列停滞。

```python
def build_lock_manager_from_env() -> WikiLockManager:

    if mode == "memory":
```

- [ ] **步骤 4：实现批次编排**

```python
class WikiIngestWorker:
    async def run_batch(self, scope: WikiScope) -> BatchResult:
        lease = await self.locks.acquire(scope.knowledge_base_id)
        if lease is None:
            if await self.store.pending_count(scope):
                raise WikiBatchBusy()
            return BatchResult()
        async with lease:
            pending = await self.store.claim_pending(
                scope, self.options.batch_size, self.options.claim_timeout
            )
            if not pending:
                if await self.store.pending_count(scope):
                    raise WikiBatchBusy()
                return BatchResult()
            mapped = await self._map_pending(scope, pending)
            update_slugs = {
                update.slug for result in mapped for update in result.updates
            }
            expected = await self.store.find_existing_pages(scope, update_slugs)
            reduced, failed_ids = await self._reduce_with_failure_isolation(
                scope,
                mapped,
                existing_pages={slug: record.page for slug, record in expected.items()},
            )
            try:
                await lease.assert_owned()
            except LockOwnershipLost as error:
                raise WikiLockLost() from error
            await self._revalidate_sources(scope, pending, failed_ids)
            try:
                await self.store.apply_results(
                    scope, pending[0].claim_token, reduced,
                    completed_op_ids=[op.id for op in pending if op.id not in failed_ids],
                    failed_op_ids=list(failed_ids),
                    expected_pages={page.slug: expected.get(page.slug) for page in reduced},
                    operation_id=uuid5(NAMESPACE_URL, f"wiki:{scope.knowledge_base_id}:{pending[0].claim_token}"),
                )
            except PageConflict:
                all_ids = [op.id for op in pending]
                await self.store.release_failed(scope, all_ids, pending[0].claim_token)
                return BatchResult.from_ids(all_ids, all_ids)
            return BatchResult.from_ids([op.id for op in pending], failed_ids)
```

Map 并发由 `asyncio.Semaphore(map_parallel)` 限制。Reduce 先按 slug 分组；任何 slug 失败时把其所有 `contributor_op_ids` 加入失败集合，删除这些 op 的全部更新并重新 Reduce 混合 slug，直到失败集合稳定。

```python
class ActivityProbe(Protocol):
    async def is_active(self, knowledge_base_id: UUID) -> bool:
        raise NotImplementedError


class WikiQueryService:
    def __init__(self, session: AsyncSession, activity_probe: ActivityProbe | None = None):
        self._session = session
        self._activity_probe = activity_probe or InactiveProbe()

    async def get_stats(self, scope: WikiScope) -> WikiStatsResponse:
        pending_tasks = await self._count(
            select(func.count(WikiPendingOp.id)).where(
                WikiPendingOp.tenant_id == scope.tenant_id,
                WikiPendingOp.knowledge_base_id == scope.knowledge_base_id,
            )
        )
        return WikiStatsResponse(
            page_count=page_count,
            folder_count=folder_count,
            link_count=link_count,
            issue_count=issue_count,
            pending_tasks=pending_tasks,
            is_active=await self._activity_probe.is_active(scope.knowledge_base_id),
        )
```

`get_wiki_services()` 通过缓存 provider 创建 Redis 或内存锁 manager，并把它作为 probe 注入 `WikiQueryService`。Redis URL 不可用时只有显式 `GRAPH_WIKI_LOCK_MODE=memory` 才允许降级。
