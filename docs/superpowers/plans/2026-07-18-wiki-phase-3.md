# Wiki 阶段三实施计划

> **面向执行代理：** 必须使用 `superpowers:subagent-driven-development`（推荐）或 `superpowers:executing-plans` 按任务执行；步骤使用复选框跟踪，并严格遵守测试先行和独立提交边界。

**目标：** 在当前 `app/` 结构中实现可追溯 citation、同范围同类型去重、贡献账本、增量重解析/删除和第 5 次失败 dead-letter，同时保持现有 REST API 合同不变。

**架构：** fake 知识来源与 fake LLM 继续提供确定性输入，PostgreSQL 保存 pending-op、页面贡献、dead-letter 和结果事务，Redis tombstone 阻止删除来源被旧 ingest 写回。Map 负责 citation、去重和贡献差量，Reduce 只重算 affected slugs，Store 在一次短事务内提交贡献、页面、链接、日志和队列状态。

**技术栈：** Python 3.12、Pydantic v2、SQLAlchemy 2 Async、PostgreSQL 16（`pg_trgm`）、Alembic、Redis、Celery、Tenacity、pytest、pytest-asyncio。

@dataclass(frozen=True, slots=True)
class PreparedCitationBatch:
    batch_index: int
    chunks: tuple[CitationBatchChunk, ...]
    alias_to_chunk_id: Mapping[str, str]


def prepare_citation_batches(
    chunks: Sequence[SourceChunk], *, max_chars: int = 12000
) -> list[PreparedCitationBatch]:
    if max_chars <= 0:
        raise ValueError("max_chars 必须大于 0")
    ordered = sorted(chunks, key=lambda item: (item.chunk_index, item.start_at, item.id))
    pieces: list[tuple[str, str]] = []
    for chunk in ordered:
        text = "\n".join(
            value.strip() for value in (chunk.text, chunk.ocr_text, chunk.image_caption)
            if value.strip()
        )
        for start in range(0, len(text), max_chars):
            pieces.append((chunk.id, text[start:start + max_chars]))

    prepared: list[PreparedCitationBatch] = []
    current: list[tuple[str, str]] = []
    size = 0
    for chunk_id, text in pieces:
        if current and size + len(text) > max_chars:
            prepared.append(_freeze_batch(len(prepared), current))
            current, size = [], 0
        current.append((chunk_id, text))
        size += len(text)
    if current:
        prepared.append(_freeze_batch(len(prepared), current))
    return prepared


def _freeze_batch(index: int, pieces: list[tuple[str, str]]) -> PreparedCitationBatch:
    aliases = [f"c{position:03d}" for position in range(len(pieces))]
    return PreparedCitationBatch(
        batch_index=index,
        chunks=tuple(
            CitationBatchChunk(alias=alias, text=text)
            for alias, (_, text) in zip(aliases, pieces, strict=True)
        ),
        alias_to_chunk_id={
            alias: chunk_id for alias, (chunk_id, _) in zip(aliases, pieces, strict=True)
        },
    )

def build_dedup_candidate_statement(
    scope: WikiScope, candidate: TopicCandidate, *, limit: int = 20
) -> Select[tuple[WikiPage]]:
    if not 1 <= limit <= 20:
        raise ValueError("dedup limit 必须在 1 到 20 之间")
    search_names = [candidate.name, *candidate.aliases]
    expression = func.lower(WikiPage.title).concat(" ").concat(
        func.lower(func.coalesce(cast(WikiPage.aliases, Text), ""))
    )
    distance = func.least(*[expression.op("<->")(name.casefold()) for name in search_names])

        select(WikiPage)
        .where(

            WikiPage.knowledge_base_id == scope.knowledge_base_id,
            WikiPage.deleted_at.is_(None),
            WikiPage.status == "published",
            WikiPage.page_type == candidate.page_type,
        )
        .order_by(distance, WikiPage.slug)
        .limit(limit)
    )

def validate_dedup_output(
    request: DedupRequest, output: DedupOutput
) -> dict[str, str | None]:
    requested = {item.candidate.slug: item for item in request.candidates}
    if len(output.decisions) != len(requested):
        raise WikiValidationError("WIKI_DEDUP_INCOMPLETE", "去重决策必须完整覆盖请求")
    resolved: dict[str, str | None] = {}
    for decision in output.decisions:
        item = requested.get(decision.candidate_slug)
        if item is None or decision.candidate_slug in resolved:
            raise WikiValidationError("WIKI_DEDUP_UNKNOWN_SOURCE", "去重来源不在请求白名单")
        allowed = {target.slug: target for target in item.allowed_targets}
        target = allowed.get(decision.canonical_slug) if decision.canonical_slug else None
        if decision.canonical_slug and target is None:
            raise WikiValidationError("WIKI_DEDUP_TARGET_NOT_ALLOWED", "去重目标不在白名单")
        if target is not None and target.page_type != item.candidate.page_type:
            raise WikiValidationError("WIKI_DEDUP_CROSS_TYPE", "去重目标类型不一致")
        if decision.canonical_slug == decision.candidate_slug:
            raise WikiValidationError("WIKI_DEDUP_SELF_TARGET", "generated slug 不能映射到自身")
        resolved[decision.candidate_slug] = decision.canonical_slug
    return resolved

def _by_slug(records: Sequence[StoredContributionRecord]) -> dict[str, StoredContributionRecord]:
    result: dict[str, StoredContributionRecord] = {}
    for record in records:
        if record.slug in result:
            raise WikiValidationError("WIKI_CONTRIBUTION_DUPLICATE_SLUG", "同一来源版本不能重复贡献 slug")
        result[record.slug] = record.model_copy(deep=True)

def plan_ingest_deltas(
    pending_op_id: UUID,
    previous: Sequence[StoredContributionRecord],
    current: Sequence[StoredContributionRecord],
) -> list[ContributionDelta]:
    old_by_slug, new_by_slug = _by_slug(previous), _by_slug(current)
    result: list[ContributionDelta] = []
    for record in current:
        old = old_by_slug.get(record.slug)
        if old is None:
            result.append(ContributionDelta(pending_op_id=pending_op_id, action="add", slug=record.slug, knowledge_id=record.knowledge_id, current=record))
        elif any(getattr(old, field) != getattr(record, field) for field in _CONTENT_FIELDS):
            result.append(ContributionDelta(pending_op_id=pending_op_id, action="replace", slug=record.slug, knowledge_id=record.knowledge_id, previous=old, current=record))
    for record in previous:
        if record.slug not in new_by_slug:
            result.append(ContributionDelta(pending_op_id=pending_op_id, action="retract_stale", slug=record.slug, knowledge_id=record.knowledge_id, previous=record))
    return result

class MemoryTombstones:
    def __init__(self, *, ttl_seconds: int = 3600, clock: Callable[[], float] = time.monotonic) -> None:
        if ttl_seconds <= 0:
            raise ValueError("ttl_seconds 必须大于 0")
        self._ttl_seconds = ttl_seconds
        self._clock = clock
        self._expires: dict[str, float] = {}

    async def mark_deleted(self, scope: WikiScope, knowledge_id: str) -> None:
        self._expires[_key(scope, knowledge_id)] = self._clock() + self._ttl_seconds

    async def is_deleted(self, scope: WikiScope, knowledge_id: str) -> bool:
        key = _key(scope, knowledge_id)
        expires = self._expires.get(key, 0)
        if expires <= self._clock():
            self._expires.pop(key, None)
            return False
        return True

def _key(scope: WikiScope, knowledge_id: str) -> str:
    normalized = knowledge_id.strip()
    if not normalized:
        raise ValueError("knowledge_id 不能为空")
    return f"wiki:deleted:{scope.knowledge_base_id}:{normalized}"


class RedisTombstones:
    def __init__(
        self, url: str, *, ttl_seconds: int = 3600, socket_timeout: float = 2.0,
        client_factory: Callable[..., Redis] = Redis.from_url,
    ) -> None:
        self._url, self._ttl_seconds = url, ttl_seconds
        self._socket_timeout, self._client_factory = socket_timeout, client_factory

    def _client(self) -> Redis:
        return self._client_factory(
            self._url, decode_responses=True,
            socket_connect_timeout=self._socket_timeout,
            socket_timeout=self._socket_timeout,
        )

    async def mark_deleted(self, scope: WikiScope, knowledge_id: str) -> None:
        client = self._client()
        try:
            await client.set(_key(scope, knowledge_id), "1", ex=self._ttl_seconds)
        finally:
            await client.aclose()

    async def is_deleted(self, scope: WikiScope, knowledge_id: str) -> bool:
        client = self._client()
        try:
            return await client.get(_key(scope, knowledge_id)) == "1"
        finally:
            await client.aclose()

```python
class WikiEnqueueService:
    def __init__(self, source: KnowledgeSourcePort, store: IngestStore, tombstones: TombstonePort) -> None:
        self.source, self.store, self.tombstones = source, store, tombstones

    async def enqueue(self, scope: WikiScope, knowledge_id: str) -> EnqueueResult:
        return await self.enqueue_ingest(scope, knowledge_id)

    async def enqueue_ingest(self, scope: WikiScope, knowledge_id: str) -> EnqueueResult:
        knowledge_id = knowledge_id.strip()
        if not knowledge_id or await self.tombstones.is_deleted(scope, knowledge_id):
            return EnqueueResult(skipped_reason="source_inactive")
        config = await self.source.get_config(scope)
        if not config.wiki_enabled:
            return EnqueueResult(skipped_reason="wiki_disabled")
        knowledge = await self.source.get_knowledge(scope, knowledge_id)
        if (
            knowledge is None
            or knowledge.id != knowledge_id
            or knowledge.tenant_id != scope.tenant_id
            or knowledge.knowledge_base_id != scope.knowledge_base_id
            or not knowledge.is_active
            or not await self.source.is_active(scope, knowledge_id, knowledge.op_version)
        ):
            return EnqueueResult(skipped_reason="source_inactive")
        chunks = await self.source.list_chunks(scope, knowledge_id)
        if not any(chunk.text.strip() or chunk.ocr_text.strip() for chunk in chunks):
            return EnqueueResult(skipped_reason="no_text_chunks")
        if not (config.synthesis_model_id or config.summary_model_id):
            return EnqueueResult(skipped_reason="model_not_configured")
        record = await self.store.enqueue_ingest(
            scope, knowledge, {"knowledge_id": knowledge_id}
        )
        return EnqueueResult(
            pending_op_id=record.pending_op_id,
            deduplicated=record.deduplicated,
        )

    async def enqueue_retract(
        self, scope: WikiScope, knowledge_id: str, op_version: str
    ) -> EnqueueResult:
        knowledge_id, op_version = knowledge_id.strip(), op_version.strip()
        if not knowledge_id or not op_version:
            return EnqueueResult(skipped_reason="invalid_retract_identity")
        await self.tombstones.mark_deleted(scope, knowledge_id)
        record = await self.store.enqueue_retract(
            scope, knowledge_id, op_version, {"knowledge_id": knowledge_id}
        )
        return EnqueueResult(pending_op_id=record.pending_op_id, deduplicated=record.deduplicated)
```

现有 `enqueue()` 只作为内部兼容别名，所有新调用使用 `enqueue_ingest()` 或 `enqueue_retract()`。

`IngestStore` 增加 `enqueue_ingest()` 和 `enqueue_retract()`；`SqlAlchemyIngestStore.enqueue_ingest()` 在插入新版本前删除同来源 `claimed_at IS NULL AND op='ingest'` 的旧 pending-op，并释放对应 finalization marker，然后幂等插入新 pending/finalization/Outbox。

`enqueue_retract()` 的同一 `session.begin()` 必须按以下代码顺序执行：

```python
# 1. 取消未 claim 的旧 ingest，并释放它们的 finalization。
# 2. 锁定当前来源 active 贡献并设为 retract_pending。
# 3. 对每个 affected slug 锁定页面和剩余 active 贡献。
# 4. 无剩余贡献时设置 deleted_at/updated_at 并删除可见链接；有剩余贡献时用 project_active_refs 更新 refs。
# 5. 只有用户可见状态或 refs 变化时 page.version += 1。
# 6. 注册 retract finalization，并幂等插入 op='retract' pending-op 与触发 Outbox。

    completed_op_ids: list[UUID] = Field(default_factory=list)
    failed_op_ids: list[UUID] = Field(default_factory=list)
    superseded_op_ids: list[UUID] = Field(default_factory=list)


def operation_failure(op_id: UUID, error: Exception) -> OperationFailure:
    if isinstance(error, PermanentModelError):
        code = "MODEL_PERMANENT"
    elif isinstance(error, TransientModelError):
        code = "MODEL_RETRY_EXHAUSTED"
    elif isinstance(error, WikiValidationError):
        code = error.code
    else:
        code = "WIKI_INGEST_FAILED"
    summary = " ".join(str(error).split())[:2000] or type(error).__name__
    return OperationFailure(pending_op_id=op_id, error_code=code[:128], error_summary=summary)
```

`BatchResult` validator必须保证三组 ID 各自唯一且互不重叠。

- [ ] **步骤 4：按 pending `op` 重构批次编排**

Worker 保留 KB 锁和 Map/Reduce semaphore，并按以下顺序执行：

1. `op='ingest'` 调用阶段三 `map_document()`；`op='retract'` 只读取该来源 `retract_pending` 贡献并调用 `plan_retract_deltas(record.id, contributions)`，不读取来源正文。
2. 不支持的 op 生成永久 `OperationFailure(code='WIKI_UNKNOWN_OP')`。
3. Map 返回 `superseded=True` 的记录进入 superseded 集合。
4. 按 delta slug 分组，读取 existing page 和当前 active contributions；每个 affected slug 只 Reduce 一次。
5. 单个 slug Reduce 失败时，标记所有贡献该 slug 的 op 为普通 failure，然后重新稳定计算剩余页面。
6. 提交前对 ingest 再调用 `source.is_active()` 与 `tombstones.is_deleted()`；失活或命中 tombstone 进入 superseded，不计失败。
7. 构造一个 `BatchApplyRequest` 并调用 store。

`TransientModelError` 仍只重试 3 次、2/4 秒。`WikiBatchBusy`、`PageConflict`、`ClaimLost`、`LockOwnershipLost`、父/子任务取消和锁丢失必须原样处理或调用 `release_claim()`，不能构造 `OperationFailure`。所有 sibling task 在异常/取消时必须 cancel 并 `gather(return_exceptions=True)`。
